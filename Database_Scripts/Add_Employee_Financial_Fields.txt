ALTER TABLE Employees

ALTER TABLE Employees

GO

-- Add Photo and License fields
ALTER TABLE Employees
ADD PhotoPath NVARCHAR(500) NULL;
GO

ALTER TABLE Employees
ADD NeedLicense BIT NOT NULL DEFAULT 1;
GO

-- Verify the changes
SELECT 
    COLUMN_NAME,
    DATA_TYPE,
    IS_NULLABLE
FROM INFORMATION_SCHEMA.COLUMNS 
WHERE TABLE_NAME = 'Employees' 
AND COLUMN_NAME IN (
    'DateOfBirth', 'MaritalStatus', 'EducationLevel',
    'CurrentSalary', 'AnnualIncreasePercentage', 'SalaryAfterAnnualIncrease',
    'BankAccountNumber', 'BankName', 'TaxId', 'InsuranceNumber',
    'PhotoPath', 'NeedLicense'
)
ORDER BY COLUMN_NAME;
GO
